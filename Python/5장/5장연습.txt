    def setdata(self, first, second):

# 사칙연산 더하기, 빼기, 곱하기, 나누기
class Fourcal:
    def setdata(self, first, second):
        self.first = first
        self.second = second
    
    def add(self):
        result = self.first + self.second
        return result

    def sub(self):
        result = self.first - self.second
        return result

    def mul(self):
        result = self.first * self.second
        return result
    
    def div(self):
        result = self.first / self.second
        return result


a = Fourcal()
b = Fourcal()
a.setdata(10,20)
b.setdata(40,50)

print(a.add())
print(a.sub())
print(a.mul())
print(a.div())

print(b.add())
print(b.sub())
print(b.mul())
print(b.div())

class Fourcal:
    def __init__(self, first, second):
        self.first = first
        self.second = second

    def setdata(self, first, second):
        self.first = first
        self.second = second
    
    def add(self):
        result = self.first + self.second
        return result

    def sub(self):
        result = self.first - self.second
        return result

    def mul(self):
        result = self.first * self.second
        return result
    
    def div(self):
        result = self.first / self.second
        return result

a = Fourcal(10,20)
print(a.first)
print(a.second)
print(a.add())
print(a.div())
print(a.mul())
print(a.sub())

b = Fourcal(20,30)
print(a.div())
